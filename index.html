Fizzbuzz with ternary syntax, array iteration methods, objects, and parameters 

        
const fizzbuzz = (values, maxNumber) => {
  for (let i = 1; i <= maxNumber; i++) {
    let output = ''
    values.forEach(item => { if (!(i % item.number)) { output += item.word } })
    output ? console.log(output) : console.log(i)
  }
}
        
        
                
// function fizzbuzz takes two parameters
// values which is an array of objects with the keys number and word
// maxNumber which is an integer representing the maxNumber value

const fizzbuzz = (values, maxNumber) => {

  // iterate from 1 to maxNumber
  for (let i = 1; i <= maxNumber; i++) {

    // declare variable output with the value of empty string
    let output = ''

    // for each item (which is an object) in the array values
    values.forEach(item => {
      // if i is divisbly by the item's number
      if (!(i % item.number)) {
        // append the item's word
        output += item.word
      }
    })
    // empty string in JavaScript evaluates to falsy
    // if output is truthy console log the output
    // else console log i
    output ? console.log(output) : console.log(i)
  }
}

// example of how the function is invoke with arguments
fizzbuzz([{number: 3, word: 'fizz'},{number: 4, word: 'rizz'}, {number: 5, word: 'buzz'}], 60)

//            @ i === 12 => 'fizzrizz'
//            @ i === 15 => 'fizzbuzz'
//            @ i === 20 => 'rizzbuzz'
//            @ i === 60 => 'fizzrizzbuzz'

ROT13 Cipher as a JavaScript Module 

            
const ROT13Cipher = {
  alphabet: function () { return 'abcdefghijklmnopqrstuvwxyzabcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZABCDEFGHIJKLMNOPQRSTUVWXYZ                                '.split('') },
  code: function (data, rotNumber = 13) { return data.split('').map(letter => this.alphabet().indexOf(letter)).map(letterIndex => this.alphabet()[letterIndex + rotNumber]).join('') },
  encode: function (data) { return this.code(data) },
  decode: function (data) { return this.code(data) }
}
            
          
                      

// ROT13Cipher is a module that has methods to encode and decode strings
// into their ROT13 values

const ROT13Cipher = {

  // key alphabet with the value of the alphabet doubled and split into an array
  // evaluates to ['a', 'b', ... , 'Y', 'Z', ... , ' ']
  alphabet: function () { return 'abcdefghijklmnopqrstuvwxyzabcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZABCDEFGHIJKLMNOPQRSTUVWXYZ                                '.split('') },
  // code function takes 2 parameters and outputs the ROT13 ciphered version
  code: function (data, rotNumber = 13) {
    // convert the input data into an array of individual letters
    return data.split('').map(letter => {
      // map each letter to the alphabet's first position of the letter
      // i.e. 'b' would be mapped to 1
      // because 'b' is at the 1th index of the alphabet array
      this.alphabet().indexOf(letter)
    }).map(letterIndex => {
      // map the index to the value of the alphabet at the letter's index + 13
      // switches the index to it's index + 13
      // and maps the number to the letter at that index
      // i.e. switches '1' to '14'
      // alphabet at position 1 is 'b'
      // alphabet at position 14 is 'o'
      this.alphabet()[letterIndex + rotNumber]
    }).join('')
  },
  encode: function (data) { return this.code(data) },
  decode: function (data) { return this.code(data) }
}

// example of usage with argument 'The quick brown fox'
ROT13Cipher.encode('The quick brown fox')

// outputs => ROT13 Cipher version of input string
//           'Gur dhvpx oebja sbk'

Tic-Tac-Toe for Any Size with Recursion 

            
const checkForWinner = (size, board) => {
  if (!board) { board = Array.from(Array(size).keys()) }
  const winner = []
  const boardSize = Math.sqrt(board.length) | 0
  const diagonal = boardSize - 1

  const checkLine = function (axis = 3, initialValue = 0, incrementX = 1, incrementY = boardSize, cycle = 1) {
    const check = []; cycle = cycle - 1
    for (let i = 0; i < boardSize; i++) { check.push(board[initialValue + incrementX * i]) }
    if (check.every(value => value === check[0])) { winner.push(check[0]) }
    if (initialValue === diagonal) { return checkLine(axis, initialValue - diagonal, diagonal + 2) }
    if (!cycle) {
      axis = axis - 1
      if (!axis && !cycle) { return }
      if (axis === 3) { return checkLine(axis, 0, incrementX, incrementY, boardSize) }
      if (axis === 2) { return checkLine(axis, 0, incrementY, incrementX, boardSize) }
      if (axis === 1) { return checkLine(axis, diagonal, diagonal) }
    }
    return checkLine(axis, initialValue + incrementY, incrementX, incrementY, cycle)
  }

  checkLine(4)
  return winner[0]
}
        
      
        
// checkForWinner checks if an array holds the same value
// for its respective virtual x-axis, y-axis and diagonal lines
// checkForWinner takes 2 argument which is an array
// size is used for testing purposes
// an array with values 0...N where N is the size is produced
// if no board array is passed in
// the array must have a length that is a square number greater than 0

const checkForWinner = (size, board) => {
  // if there is no board, produce one with the values 0 to N
  // i.e. size 16 produces board = [0, 1, 2, ..., 15]
  if (!board) { board = Array.from(Array(size).keys()) }

  // declare variable winner which is an array which will hold the winner (X or O)
  const winner = []

  // declare variable boardSize which is the length of the side of a board
  // i.e. board of size 16 is an abstraction of a 4x4 grid
  // the length of that side is the square root of the length of the size
  const boardSize = Math.sqrt(board.length) | 0

  // diagonal top right to bottom left starts at boardSize - 1
  const diagonal = boardSize - 1

  // recursive function that takes 5 paramters
  // initialValue is the start of the board (top left)
  // incrementX is the difference on the X axis from one index to the next
  // incrementY is the difference on the Y axis from one index to the next
  // cycle is the number of times the function is run per axis (X, Y, diagonal)
  const checkLine = function (axis = 3, initialValue = 0, incrementX = 1, incrementY = boardSize, cycle = 1) {

    // declare variable check which will hold the values
    const check = []

    // de-increment cycle on each recursive call
    cycle = cycle - 1

    // alias axis numbers to their representation
    const axes = { 3: 'X axis', 2: 'Y axis', 1: 'Diagonal' }

    // loop through the board size
    for (let i = 0; i < boardSize; i++) {
      // push onto check array the value of board at position
      // on first recursive call in 4x4 board: pushes values at index 0, 1, 2, 3
      // on second recursive call in 4x4 board: pushes values at index 4, 5, 6, 7
      // on fifth recursive call in 4x4 board: pushes values at index 0, 4, 8, 12
      check.push(board[initialValue + incrementX * i])
     }

    // for testing purposes
    if (axis <= 3) {
      // console log the axis and indexes that are checked
      console.log(`${axes[axis]} indexes: ${check}`)
     }

    // if every value in the array is the same as the first one
    if (check.every(value => value === check[0])) {
      // push the value onto the winner array
      winner.push(check[0])
    }

    // if initialValue starts at the index of the top-right to bottom-left diagonal
    if (initialValue === diagonal) {
      // flip to top-left to bottom-right diagonal
      return checkLine(axis, initialValue - diagonal, diagonal + 2)
    }
    // if cycle is falsy (reaches 0)
    if (!cycle) {
      // de-increment axis
      axis = axis - 1
      // if axis and cycle are both falsy (both 0)
      if (!axis && !cycle) {
        // exit recursion
        return
      }

      if (axis === 3) {
        // check all values in X axis direction
        return checkLine(axis, 0, incrementX, incrementY, boardSize)
       }
      if (axis === 2) {
        // check all values in Y axis direction
        return checkLine(axis, 0, incrementY, incrementX, boardSize)
       }
      if (axis === 1) {
        // check all values in top-right to bottom-left diagonal
        return checkLine(axis, diagonal, diagonal)
       }
    }
    return checkLine(axis, initialValue + incrementY, incrementX, incrementY, cycle)
  }

  // invoke recursive function
  checkLine(4)

  // return the value at position 0 of winner array
  return winner[0]
}
